11/** @page synchronized_iterator Synchronized Iterator
12 * Two classes that provide "synchronized" iterators through a vector of ordered vectors,
13 * (or of some ordered OrdContainer that have a similar iterator),
14 * needed in computation of post
15 * in subset construction, product, and non-determinization.
16 *
17 * The Type stored in OrdContainers must be comparable with `<`,`>`,`==`,`!=`,`<=`,`>=`,
18 * and it must be a total (linear) ordering.
19 * The intended usage in, for instance, determinisation is for Type to be TransSymbolStates.
20 * TransSymbolStates is ordered by the symbol.
21 *
22 * SynchronisedIterator is the parent virtual class.
23 * It stores a vector of end-iterators for the OrdContainer v and a vector of current positions.
24 * They are filled in using the function push_back(begin,end), that adds begin and end iterators of v to positions and
25 * ends, respectively
26 * Method advance advances all positions forward so that they are synchronized on the next smallest equiv class
27 * (next smallest symbol in the case of TransSymbolStates).
28 *
29 * There are two versions of the class.
30 * i) In product, ALL positions must point to currently the smallest equiv. class (Moves with the same symbol).
31 * Method get_current then returns the vector of all position iterators, synchronized.
32 * ii) In determinization, it is enough that there EXISTS a position that points to the smallest class.
33 * Method get_current then returns the vector of only those positions that point to the smallest equiv. class.
34 *
35 * Usage: 0) construct, 1) fill in using @c SynchronizedIterator::push_back(), iterate using @c SynchronizedIterator::advance() and @c SynchronizedIterator::get_current(), 2) @c SynchronizedIterator::reset()
36 * and go back to 1).
37 *
38 * @note The memory allocated internally for positions and ends is kept after @c SynchronizedIterator::reset(), so it is advisable to use the
39 * same iterator for many iterations, as opposed to creating a new one for each iteration.
40 */

93template<typename Iterator>
94class SynchronizedUniversalIterator: public SynchronizedIterator<Iterator> {
95public:
96
97 /** "minimum" would be the smallest class bounded from below by all positions that appears in all OrdContainers.
98 * Are we sure that all positions at this class?
99 * Invariant: it can be true only if all positions are indeed synchronized.
100 */
101 bool synchronized_at_current_minimum = false;
102
103 /**
104 * Advances all positions to the NEXT minimum and returns true (though the next minimum might be the current state
105 * if synchronized_at_current_minimum is false), or returns false if positions cannot be synchronized.
106 *
107 * If positions are synchronized to start with, then synchronized_at_current_minimum decides whether to stay or
108 * advance further.
109 * The general of the algorithm is to synchronize everybody with position[0].
110 */
111 bool advance() {
112 //Nothing to synchronize.
113 if (this->positions.empty()) { return false; } // TODO: ?? or not?
114
115 // If already synchronized, start moving forward by advancing position[0] (and so break synchronization).
116 if (this->synchronized_at_current_minimum) {
117 ++this->positions[0];
118 synchronized_at_current_minimum = false;
119 }
120
121 // If positions[0] has nowhere to go, then sync is not possible.
122 if (this->positions[0] == this->ends[0]) { return false; }
123
124 // Synchronise all positions with position[0].
125 for (size_t i = 1, positions_size = this->positions.size(); i < positions_size; ++i) {
126 // If some positions has nowhere to go, then sync is not possible.
127 if (this->positions[i] == this->ends[i]) { return false; }
128
129 // Advance position[i] and position[0] to the closest equal values.
130 while (*this->positions[i] != *this->positions[0]) {
131
132 // Advance position[i] to or beyond position[0].
133 while (*this->positions[i] < *this->positions[0]) {
134 ++this->positions[i];
135 if (this->positions[i] == this->ends[i]) { return false; }
136 }
137
138 // Advance position[0] to or beyond position[i].
139 while (*this->positions[i] > *this->positions[0]) {
140 ++this->positions[0];
141 if (this->positions[0] == this->ends[0]) { return false; }
142 }
143
144 // If position[0] changed, start from position 1 again.
145 // (note that
146 // i gets incremented at the end of the for-loop body,
147 // and that,
148 // since we are inside the for, there are at least two positions
149 // as the for starts with i=1.)
150 if (this->positions[0] > this->positions[1]) { i=0; }
151 }
152 }
153 this->synchronized_at_current_minimum = true;
154 return true;
155 }
156
157 /// Returns the vector of current positions.
158 const std::vector<Iterator>& get_current() const {
159 return this->positions;
160 };
161
162 explicit SynchronizedUniversalIterator(const size_t size=0) : SynchronizedIterator<Iterator>(size) {};
163
164 void reset(const size_t size = 0) {
165 SynchronizedIterator < Iterator > ::reset(size);
166 this->synchronized_at_current_minimum = false;
167 };
168}; // class SynchronizedUniversalIterator.

170template<typename Iterator>
171class SynchronizedExistentialIterator : public SynchronizedIterator<Iterator> {
172public:
173 Iterator get_current_minimum() {
174 if (currently_synchronized.empty()) {
175 throw std::runtime_error("Trying to get minimum from sync. ex. iterator which has no minimum. Don't do "
176 "that ever again or your nose falls off!");
177 }
178 return currently_synchronized[0];
179 }
180
181 std::vector<Iterator> currently_synchronized{}; // Positions that are currently synchronized.
182 Iterator next_minimum{}; // The value we should synchronise on after the first next call of advance().
183
184 bool is_synchronized() const { return !currently_synchronized.empty(); }
185
186 /**
187 * Advances all positions just above current_minimum,
188 * that is, to or above next_minimum.
189 * Those at next_minimum are added to currently_synchronized.
190 * Since next_minimum becomes the current minimum,
191 * new next_minimum must be updated too.
192 */
193 bool advance() override {
194 // The next_minimum becomes the current current_minimum.
195 auto current_minimum = this->next_minimum;
196
197 // Here we collect the result.
198 currently_synchronized.clear();
199
200 for (size_t i = 0, positions_size = this->positions.size(); i < positions_size;) {
201 Iterator& position_i_it{ this->positions[i] };
202 Iterator& end_i_it{ this->ends[i] };
203 if (position_i_it == end_i_it) {
204 // If there is nothing left at the position, it is removed, that is,
205 // swapped with a position from the end of the vector,
206 // and the vector is shortened.
207 // The same is done with ends.
208 while (position_i_it == end_i_it && i < positions_size) {
209 position_i_it = this->positions[positions_size - 1];
210 end_i_it = this->ends[positions_size - 1];
211 // This must be here, because in the next call, position_size is set again with positions.size().
212 this->positions.pop_back();
213 this->ends.pop_back();
214 --positions_size;
215 }
216 }
217 // If the i-th position, i.e., where we are now, was erased,
218 // then we reached the end of active positions.
219 if (i == positions_size) { return !currently_synchronized.empty(); }
220
221 // If we are at the current_minimum, then save it and advance the position.
222 if (*position_i_it == *current_minimum) {
223 this->currently_synchronized.emplace_back(position_i_it);
224 ++position_i_it;
225 continue;
226 }
227
228 // If the position (now larger than current_minimum) is smaller than next_minimum,
229 // or next_minimum has not yet been updated, then update the next_minimum.
230 if (*this->next_minimum > *position_i_it || *this->next_minimum == *current_minimum) {
231 this->next_minimum = position_i_it;
232 }
233
234 ++i; // This cannot be in the for statement line, because of the continue in the if body above.
235 }
236 return !currently_synchronized.empty();
237 }; // advance().
238
239 /**
240 * @brief Returns the vector of current still active positions.
241 *
242 * Beware, they will be ordered differently from how there were input into the iterator.
243 * This is due to swapping of the emptied positions with positions at the end.
244 */
245 const std::vector<Iterator>& get_current() const override { return this->currently_synchronized; };
246
247 void push_back(const Iterator &begin, const Iterator &end) override {
248 // Empty vector would not have any effect (unlike in the case of the universal iterator).
249 if (begin == end) return;
250
251 // Initialise next_minimum as the first position at the first vector.
252 if (this->positions.empty()) {
253 this->next_minimum = begin;
254 } else if (*this->next_minimum > *begin) {
255 // If the first position is of the new vector is smaller than minimum, update minimum.
256 this->next_minimum = begin;
257 }
258
259 // Let position point to the beginning the vector,
260 // save the end of the vector.
261 this->positions.emplace_back(begin);
262 this->ends.emplace_back(end);
263 }
264
265 explicit SynchronizedExistentialIterator(const size_t size=0) : SynchronizedIterator<Iterator>(size) {
266 this->currently_synchronized.reserve(size);
267 }
268
269 void reset(const size_t size = 0) {
270 SynchronizedIterator<Iterator>::reset(size);
271 if (size > 0) {
272 this->currently_synchronized.reserve(size);
273 }
274 this->currently_synchronized.clear();
275 }
276};
