21/**
22 * A node of graph representing transition formula. A node could be operator (!,&,|) or operand (symbol, state, node).
23 * Each node has a name (in case of marking naming, an initial character defining type of node is removed and stored in
24 * name), raw (name including potential type marker), and information about its type.

106/**
107 * Structure representing a transition formula using a graph.
108 * A node of graph consists of node itself and set of children nodes.
109 * Nodes are operators and operands of the formula.
110 * E.g., a formula q1 & s1 will be transformed to a tree with & as a root node
111 * and q1 and s2 being children nodes of the root.
112 */
113class FormulaGraph {
114private:
115 /// Maximal number of @c FormulaNode children in any @c FormulaGraph node (for AND, OR nodes). Only one (NOT node)
116 /// or zero (@c FormulaNode node) children may be used. Used as an optimalization by preallocating @c children at
117 /// node initialization.
118 static constexpr size_t MAX_NUM_OF_CHILDREN{ 2 };
119public:
120 FormulaNode node{};
121 std::vector<FormulaGraph> children{};
122
123 FormulaGraph() = default;
124 explicit FormulaGraph(const FormulaNode& n) : node(n), children() { children.reserve(MAX_NUM_OF_CHILDREN); }
125 explicit FormulaGraph(FormulaNode&& n) : node(std::move(n)), children() { children.reserve(MAX_NUM_OF_CHILDREN); }
126 FormulaGraph(const FormulaGraph& g) : node(g.node), children(g.children) {}
127 FormulaGraph(FormulaGraph&& g) noexcept : node(std::move(g.node)), children(std::move(g.children)) {}
128
129 FormulaGraph& operator=(const mata::FormulaGraph&) = default;
130 FormulaGraph& operator=(mata::FormulaGraph&&) noexcept = default;
131
132 std::unordered_set<std::string> collect_node_names() const;
133 void print_tree(std::ostream& os) const;
134};

136/**
137 * Structure for a general intermediate representation of parsed automaton.
138 * It contains information about type of automata, type of naming of nodes, symbols and states
139 * and type of alphabet. It contains also the transitions formula and formula for initial and final
140 * states. The formulas are represented as tree where nodes are either operands or operators.
141 */
142class IntermediateAut {
143public:
144 /**
145 * Type of automaton. So far we support nondeterministic finite automata (NFA) and
146 * alternating finite automata (AFA)
147 */
148 enum class AutomatonType {
149 NFA,
150 AFA,
151 NFT
152 };
153
154 /**
155 * The possible kinds of naming of items in sets of states, nodes, or symbols.
156 * It implies how the given set is defined.
157 * Naming could be automatic (all things in formula not belonging to other sets will be assigned to a
158 * set with automatic naming), marker based (everything beginning with `q` is a state, with `s` is a symbol,
159 * with `n` is a node), or enumerated (the given set is defined by enumeration).
160 * There are two special cases used for alphabet - symbols could be any character (CHARS) or anything from utf (UTF).
161 */
162 enum class Naming {
163 AUTO,
164 MARKED,
165 ENUM,
166 CHARS,
167 UTF
168 };
169
170 /**
171 * Type of alphabet. We can have explicit alphabet (containing explicit symbols), or bitvector, or class of symbols
172 * (e.g., alphabet is everything in utf), or intervals.
173 * So far, only explicit representation is supported.
174 */
175 enum class AlphabetType {
176 EXPLICIT,
177 BITVECTOR,
178 CLASS,
179 INTERVALS
180 };
181
182 Naming state_naming = Naming::MARKED;
183 Naming symbol_naming = Naming::MARKED;
184 Naming node_naming = Naming::MARKED;
185 AlphabetType alphabet_type{};
186 AutomatonType automaton_type{};
187
188 // The vectors represent the given sets when enumeration is used.
189 std::vector<std::string> states_names{};
190 std::vector<std::string> symbols_names{};
191 std::vector<std::string> nodes_names{};
192
193 FormulaGraph initial_formula{};
194 FormulaGraph final_formula{};
195
196 bool initial_enumerated = false;
197 bool final_enumerated = false;
198
199 /**
200 * Transitions are pairs where the first member is left-hand side of transition (i.e., a state)
201 * and the second item is a graph representing transition formula (which can contain symbols, nodes, and states).
202 */
203 struct std::vector<std::pair<FormulaNode, FormulaGraph>> transitions{};

205 /**
206 * Returns symbolic part of transition. That may be just a symbol or bitvector formula.
207 * This function is supported only for NFA where transitions have a rhs state at the end of right
208 * handed side of transition.
209 * @param transition Transition from which symbol is returned
210 * @return Graph representing symbol. It maybe just an explicit symbol or graph representing bitvector formula
211 */
212 const FormulaGraph& get_symbol_part_of_transition(const std::pair<FormulaNode, FormulaGraph>& trans) const;

218 * It parses basic information about type of automata, its naming conventions etc.

226 static std::vector<IntermediateAut> parse_from_mf(const mata::parser::Parsed& parsed);
