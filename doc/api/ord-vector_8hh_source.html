17template <class T>
18bool are_disjoint(const utils::OrdVector<T>& lhs, const utils::OrdVector<T>& rhs) {
19 auto itLhs = lhs.begin();
20 auto itRhs = rhs.begin();
21 while (itLhs != lhs.end() && itRhs != rhs.end()) {
22 if (*itLhs == *itRhs) { return false; }
23 else if (*itLhs < *itRhs) { ++itLhs; }
24 else {++itRhs; }
25 }
26 return true;
27}

29template <class Key>
30bool is_sorted(const std::vector<Key>& vec) {
31 for (auto itVec = vec.cbegin() + 1; itVec < vec.cend(); ++itVec) {
32 if (!(*(itVec - 1) < *itVec)) {
33 // In case there is an unordered pair (or there is one element twice).
34 return false;
35 }
36 }
37 return true;

395 bool is_intersection_empty_with(const OrdVector& rhs) const {
396 assert(is_sorted());
397 assert(rhs.is_sorted());
398
399 const_iterator itLhs = begin();
400 const_iterator itRhs = rhs.begin();
401
402 while ((itLhs != end()) && (itRhs != rhs.end()))
403 { // until we drop out of the array (or find a common element)
404 if (*itLhs == *itRhs)
405 { // in case there exists a common element
406 return false;
407 }
408 else if (*itLhs < *itRhs)
409 { // in case the element in lhs is smaller
410 ++itLhs;
411 }
412 else
413 { // in case the element in rhs is smaller
414 assert(*itLhs > *itRhs);
415 ++itRhs;
416 }
417 }
418 return true;
419 }

424 static OrdVector difference(const OrdVector& lhs, const OrdVector& rhs) {
425 assert(lhs.is_sorted());

427
428 OrdVector result{};

433 if (rhs_it == rhs.end()) {
434 result.push_back(*lhs_it);

443 ++rhs_it;
