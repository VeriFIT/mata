35/**
36 * @brief Implementation of a set of non-negative numbers using sparse-set.
37 *
38 * This class implements the interface of a set (similar to std::set) using sparse-set date structure,
39 * that is, a pair of vectors dense and sparse (... google it).
40 * Importantly
41 * - Insertion and removal are constant time.
42 * - Iteration is linear in the number of stored elements.
43 * - It takes a lot of space, the sparse and dense vectors allocate as many indexes as the maximal stored number.
44 *
45 * @tparam Number Number type: type of numbers contained in the container.
46 */
47 template<typename Number>
48 class SparseSet {
49 static_assert(std::is_unsigned<Number>::value, "SparseSet can only contain unsigned integers");
50
51 private:
52 std::vector<Number> dense{}; // Dense set of elements.
53 std::vector<Number> sparse{}; // Map of elements to dense set indices.
54
55 /// Number of elements which are in the set (current size).
56 size_t size_ = 0;
57
58 /// @brief Over-approximation of Number in sparse set. It represents the Numbers which were in the set throughout the
59 /// life of the set until now.
60 ///
61 /// `truncate()` will update the domain size to the current maximal Number + 1.
62 /// Constructor can preallocate the sparse set and set domain size to the requested value.
63 /// The structures are preallocated to at least @c domain_size_ size (can be more when maximal Number is removed).
64 size_t domain_size_ = 0;
65
66 public:
67 using iterator = typename std::vector<Number>::const_iterator;
68 using const_iterator = typename std::vector<Number>::const_iterator;
69
70 iterator begin() { return dense.begin(); }
71
72 const_iterator begin() const { return dense.begin(); }
73
74 iterator end() { return dense.begin() + static_cast<long>(size_); }
75
76 const_iterator end() const { return dense.begin() + static_cast<long>(size_); }
77
78 size_t size() const { return size_; }
79 size_t domain_size() const { return domain_size_; }
80
81 bool empty() const { return size_ == 0; }
82
83 void clear() { size_ = 0; }
84
85 // TODO: maybe we could reserve space more efficiently, by something as doubling?
86 // But we should not create havoc with domain_size, which is used outside, namely for determining the states of an automaton.
87 void reserve(size_t u) {
88 if (u > domain_size_) {
89 dense.resize(u, 0);
90 sparse.resize(u, 0);
91 domain_size_ = u;
92 }
93 assert(consistent());
94 }
95
96 bool contains(const Number val) const {
97 return val < domain_size_ &&
98 sparse[val] < size_ &&
99 dense[sparse[val]] == val;
100 }
101
102 void insert(const Number val) {
103 assert(consistent());
104
105 if (!contains(val)) {
106 if (static_cast<size_t>(val) >= domain_size_) {
107 reserve(val + 1);
108 }
109 dense[size_] = val;
110 sparse[val] = static_cast<Number>(size_);
111 ++size_;
112 }
113
114 assert(consistent());
115 }
116
117 /**
118 * Erase @p number from the set without checking for its existence.
119 * @param number Number to be erased.
120 * @pre @p number exists in the set.
121 */
122 void erase_nocheck(const Number number) {
123 dense[sparse[number]] = dense[size_ - 1];
124 sparse[dense[size_ - 1]] = sparse[number];
125 --size_;
126 }
127
128 void erase(const Number val) { if (contains(val)) { erase_nocheck(val); } }

200 bool consistent() {
201 return (domain_size_ >= size_ &&
202 (max() < domain_size_ || (size_ == 0 && domain_size_ == 0) ) &&
203 dense.size() >= domain_size_ &&
204 sparse.size() >= domain_size_

257 /*
258 * Complements the set with respect to a given number of elements = the maximum number + 1.
259 */
260 void complement(Number new_domain_size) {
261 Number old_domain_size = domain_size_;
262 for (Number i = 0; i < new_domain_size; ++i) {
263 if (contains(i))
264 erase_nocheck(i);
265 else
266 insert(i);
267 }
268 for (Number i = new_domain_size; i < old_domain_size; ++i) {
269 erase(i);
270 }
271
272 assert(consistent());
273 }

275 template<typename F>
276 void filter(F && is_staying) {
277 for (Number i{ 0 }; i < size_; ++i) {
278 while (i < size_ && !is_staying(dense[i])) {
279 erase(dense[i]);
280 }
281 }

283 assert(consistent());
284 }

286 void sort() {
287 std::sort(dense.begin(), dense.begin() + static_cast<long>(size_));
288 for (Number i = 0; i < size_; i++) {
289 sparse[dense[i]] = i;
290 }
291
292 assert(consistent());
293 }

295 template<typename F>
296 void rename(F && renaming) {
297 for (Number i = 0; i < size_; i++) {
298 if (dense[i] != renaming(dense[i])) {
299 dense[i] = renaming(dense[i]);
300 if (dense[i] >= domain_size_)
301 reserve(dense[i]+1);
302 sparse[dense[i]] = i;
303 }
304 }
305
306 assert(consistent());
307 }

309 /// @brief Maximal Number in set.
310 ///
311 /// Expensive operation as it has to compute the maximal Number in linear time.
312 Number max() {
313 Number max = 0;
314 for (Number i = 0;i<size_;i++) {
315 if (max < dense[i])
316 max = dense[i];
317 }
318 return max;
319 }
320
321 void truncate() {
322 if (size_ == 0)
323 domain_size_ = 0;
324 else
325 domain_size_ = max() + 1;
326
327 assert(consistent());
328 }
