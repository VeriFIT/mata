1/**
2 * @file closed_set.hh
3 * @brief Definition of a downward and upward closed set.
4 *
5 * This file contains definition of a closed set and function which
6 * allow us to work with them (membership, inclusion, union, intersection)
7 *
8 * Description:
9 * In this context, an upward-closed set is a set of sets of elements of type T,
10 * (set of nodes of type T, where a node is a set of elements of type T)
11 * where the elements come from the discrete range <min_val; max_val>. If the upward
12 * closed set contains a subset A of the range <min_val; max_val>, it also contains
13 * all the supersets of A (in context of the discrete range <min_val; max_val>).
14 * Thus, the upward closed set could be easily represented by its 1) type (upward closed),
15 * 2) discrete range borders (min_val, max_val), 3) its antichain.
16 * Analogously, a downward closed set contains all the subsets of the antichain elements.
17 *
18 * It is possible to:
19 *
20 * -> ==- and !=-compare two closed sets
21 * -> <=- and >=-compare two closed sets of the same type
22 * -> get a text representation of the closed set
23 * -> chceck whether the closed set contains a given node/nodes
24 * -> insert a node/more nodes to the closed set
25 * -> perform an union over two closed sets of the same type and with the same carrier
26 * -> perform an intersection over two closed sets of the same type and with the same carrier
27 * -> compute a complement of a closed set
28 *
29 * It is not possible to:
30 *
31 * -> choose a custom carrier which is not a discrete range <min_val; max_val>
32 * -> <=- and >=-compare two closed sets of the different types (nonsense)
33 * -> remove a node/more nodes from the closed set (nonsense)
34 * -> perform an union over two closed sets of different types or different carriers (nonsense)
35 * -> perform an intersection over two closed sets of different types or different carriers (nonsense)
36 *
37 * Examples:
38 *
39 * Let us have the carrier {0, 1, 2, 3} and the upward-closed set with the antichain
40 * {{0}, {1, 2}}. We can write↑{{0}, {1, 2}} = {{0}, {0, 1}, {0, 2}, {0, 3}, {0, 1, 2},
41 * {0, 1, 3}, {0, 2, 3}, {1, 2}, {1, 2, 3}, {0, 1, 2, 3}}.
42 *
43 * Let us have the carrier {0, 1, 2, 3} and the downward-closed set with the antichain
44 * {{0}, {1, 2}}. We can write ↓{{0}, {1, 2}} = {{0}, {1, 2}, {1}, {2}, {}}.
45 *
46 * @author Tomáš Kocourek
47 */

190/** This function decides whether a set of elements is part of the closed set
191* by subset-compraring the input with all elements of the antichain
192* @brief decides whether the closed set contains a given element
193* @param node a given ordered vector of elements of the type T
194* @return true iff the given ordered vector belongs to the current closed set
195*/
196template <typename T>
197bool ClosedSet<T>::contains(const Node& node) const {
198 if(type_ == ClosedSetType::upward_closed_set) {
199 for(auto element : antichain_) {
200 if(element.is_subset_of(node)) {
201 return true;
202 }
203 }
204 }
205 else if(type_ == ClosedSetType::downward_closed_set) {
206 for(auto element : antichain_) {
207 if(node.is_subset_of(element)) {
208 return true;
209 }
210 }
211 }
212 return false;
213}

215/** This function decides whether a set of sets of elements is a part of the closed set
216* by subset-compraring the input with all elements of the antichain
217* @brief decides whether the closed set contains a given set of sets of elements

219* @return true iff the given ordered vector of ordered vectors belongs to the current closed set

221template <typename T>
222bool ClosedSet<T>::contains(const Nodes& nodes) const {
223 for(auto node : nodes) {

231/** This function decides whether a given ordered vector of elements of the datatype T
232* belongs to the discrete interval of the current closed set

237template <typename T>
238bool ClosedSet<T>::in_interval(const Node& node) const {
239 for(auto value : node) {
240 if(value < min_val_ || value > max_val_) {
241 return false;

244 return true;

247/** Adds a new element to the closed set. If the element is already contained in the closed
248* set, the closed set remains unchanged. Otherwise, the antichain will be recomputed.
249* @brief inserts a new element to the closed set
250* @param node a given node which will be added to the closed set
251*/
252template <typename T>
253void ClosedSet<T>::insert(const Node& node) {
254 assert(in_interval(node) && "Each element of the given node has to respect " &&
255 "the carrier of the closed set.");
256 // If the closed set is empty, the antichain could be simply changed
257 // by adding the given node as the only node to the antichain
258 if(antichain_.empty()) {
259 antichain_.insert(node);
260 return;
261 }
262 // If the closed set already contains the given node, there is no
263 // need to change the closed set
264 if(contains(node)) {
265 return;
266 }
267 // We need to collect all the elements off the antichain which could be removed
268 // as soon as the new element is added to keep the antichain <=-uncomparable
269 OrdVec<OrdVec<T>> to_erase{};
270
271 // If the closed set is upward-closed, we have to erase all the elements of
272 // the antichain which are supersets of the inserted node
273 // Example: Let us have an upward-closed set ↑{{0, 1}, {2}} with a corresponding
274 // antichain {{0, 1}, {2}}. If we add {0} to the closed set, the element {0, 1}
275 // needs to be erased from the antichain since the set {{0}, {0, 1}, {2}} contains
276 // <=-comparable elements and the result should be upward-closed.
277 if(type_ == ClosedSetType::upward_closed_set) {
278 for(auto element : antichain_) {
279 if(node.is_subset_of(element)) {
280 to_erase.insert(element);
281 }
282 }
283 }
284
285 // If the closed set is downward-closed, we have to erase all the elements of
286 // the antichain which are subsets of the inserted node
287 // Example: Let us have an upward-closed set ↓{{0, 1}, {2}} with a corresponding
288 // antichain {{0, 1}, {2}}. If we add {1, 2} to the closed set, the element {2}
289 // needs to be erased from the antichain since the set {{0}, {1, 2}, {2}} contains
290 // <=-comparable elements and the result should be downward-closed.
291 else if(type_ == ClosedSetType::downward_closed_set) {
292 for(auto element : antichain_) {
293 if(element.is_subset_of(node)) {
294 to_erase.insert(element);
295 }
296 }
297 }
298
299 for(auto element : to_erase) {
300 antichain_.erase(element);
301 }
302 antichain_.insert(node);
303} // insert }}}

305/** Performs an union over two closed sets with the same type and carrier.
306* This function simply adds all the
307* elements of the antichain of the first closed set to the second closed set
308* @brief performs an union over closed sets
309* @param rhs a given closed set which will be unioned with the current one
310* @return an union of the given closed sets
311*/
312template <typename T>
313ClosedSet<T> ClosedSet<T>::set_union(const ClosedSet<T>& rhs) const {
314 assert(type_ == rhs.type_ && min_val_ == rhs.min_val_ && max_val_ == rhs.max_val_ &&
315 "Types and borders of given closed sets must be the same to compute their union.");
316 ClosedSet<T> result(type_, min_val_, max_val_, antichain_);
317 result.insert(rhs.antichain());
318 return result;
319}

321/** Performs an intersection over two closed sets with the same type and carrier.
322* @brief performs an intersection over closed sets
323* @param rhs a given closed set which will be intersectioned with the current one
324* @return an intersection of the given closed sets
325*/
326template <typename T>
327ClosedSet<T> ClosedSet<T>::intersection(const ClosedSet<T>& rhs) const {
328 assert(type_ == rhs.type_ && min_val_ == rhs.min_val_ && max_val_ == rhs.max_val_ &&
329 "Types and borders of given closed sets must be the same to compute their union.");
330 ClosedSet<T> result(type_, min_val_, max_val_);
331
332 // Iterates through all the tuples from Antichan1 X Antichan2
333 // and creates an union of them
334 if(type_ == ClosedSetType::upward_closed_set) {
335 for(const Node& element1 : antichain_) {
336 for(const Node& element2 : rhs.antichain()) {
337 Node tmp = element1;
338 tmp.insert(element2);
339 result.insert(tmp);
340 }
341 }
342 }
343
344 // Iterates through all the tuples from Antichan1 X Antichan2
345 // and creates an intersection of them
346 if(type_ == ClosedSetType::downward_closed_set) {
347 for(auto element1 : antichain_) {
348 for(auto element2 : rhs.antichain()) {
349 result.insert(element1.intersection(element2));
350 }
351 }
352 }
353 return result;
354} // intersection }}}

356/** Performs a complementation over a closed set. The result will
357* contain nodes which are not elements of the former closed set.
358* The complement of an upward-closed set is always downward-closed and vice versa.
359* @brief performs a complementation over a closed set
360* @return a complement of a closed set
361*/
362template <typename T>
363ClosedSet<T> ClosedSet<T>::complement() const {
364 // The complement of an upward-closed set is
365 // always downward-closed and vice versa.
366 ClosedSetType new_type = ClosedSetType::upward_closed_set;
367 if(type_ == ClosedSetType::upward_closed_set) {
368 new_type = ClosedSetType::downward_closed_set;
369 }
370 ClosedSet<T> result = ClosedSet(new_type, get_min(), get_max(), antichain());
371
372 if(type_ == ClosedSetType::upward_closed_set) {
373 // Initially, a complement contains all possible elements
374 // which will be then (possibly) removed.
375 Node initialValues{};
376 for(long unsigned i = 0; i <= max_val_; ++i) {
377 initialValues.insert(i);
378 }
379 result.insert(initialValues);
380
381 // For each element of the closed set {xa, xb, xc, ...}, we
382 // create a set of all sets Xa, Xb, Xc,... such that Xn contains
383 // all elements of the carrier except of xn
384 // For example, for a node {0, 2, 3} (the maximal element is 4),
385 // we create a set {{1, 2, 3, 4}, {0, 1, 3, 4}, {0, 1, 2, 4}}.
386 // This set corresponds to an antichain of a new downward-closed set.
387 // The result will be an intersection of all downward-closed sets
388 // created using this procedure.
389 for(const auto& element : antichain()) {
390 ClosedSet preparingAntichain(ClosedSetType::downward_closed_set, get_min(), get_max());
391 for(T i = 0; i <= max_val_; ++i) {
392 if(!element.count(i)) {
393 continue;
394 }
395 Node candidates{};
396 for(T j = 0; j <= max_val_; ++j) {
397 if(i != j) {
398 candidates.insert(j);
399 }
400 }
401 preparingAntichain.insert(candidates);
402 }
403 result = result.intersection(preparingAntichain);
404 }
405 }
406
407 else if(type_ == ClosedSetType::downward_closed_set) {
408 // Initially, a complement contains all possible elements
409 // which will be then (possibly) removed.
410 result.insert(Node());
411
412 // For each element of the closed set {xa, xb, xc, ...}, we
413 // create a set of all sets Xa, Xb, Xc,... such that Xn contains
414 // only xn. For example, for a node {0, 2, 3}, we create a set
415 // {{0}, {2}, {3}}.
416 // This set corresponds to an antichain of a new upward-closed set.
417 // The result will be an intersection of all upward-closed sets
418 // created using this procedure.
419 for(Node element : antichain()) {
420 ClosedSet preparingAntichain(ClosedSetType::upward_closed_set, min_val_, max_val_);
421 for(T i = min_val_; i <= max_val_; ++i) {
422 Node candidates{i};
423 if(!element.count(i)) {
424 preparingAntichain.insert({i});
425 }
426 }
427 result = result.intersection(preparingAntichain);
428 }
429 }
430
431 return result;
432} // complement }}}
