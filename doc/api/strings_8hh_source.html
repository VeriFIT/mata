28class ShortestWordsMap {
29public:
30 /**
31 * Maps states in the automaton @p aut to shortest words accepted by languages of the states.
32 * @param aut Automaton to compute shortest words for.
33 */
34 explicit ShortestWordsMap(const Nfa& aut) : reversed_automaton(revert(aut)) {
35 insert_initial_lengths();
36 compute();
37 }
38
39 /**
40 * Gets shortest words for the given @p states.
41 * @param[in] states States to map shortest words for.
42 * @return Set of shortest words.
43 */
44 std::set<Word> get_shortest_words_from(const StateSet& states) const;
45
46 /**
47 * Gets shortest words for the given @p state.
48 * @param[in] state State to map shortest words for.
49 * @return Set of shortest words.
50 */
51 std::set<Word> get_shortest_words_from(State state) const;
52
53private:
54 using WordLength = int; ///< A length of a word.
55 /// Pair binding the length of all words in the word set and word set with words of the given length.
56 using LengthWordsPair = std::pair<WordLength, std::set<Word>>;
57 /// Map mapping states to the shortest words accepted by the automaton from the mapped state.
58 std::unordered_map<State, LengthWordsPair> shortest_words_map{};
59 std::set<State> processed{}; ///< Set of already processed states.
60 std::deque<State> fifo_queue{}; ///< FIFO queue for states to process.
61 const Nfa reversed_automaton; ///< Reversed input automaton.
62
63 /**
64 * @brief Inserts initial lengths into the shortest words map.
65 *
66 * Inserts initial length of length 0 for final state in the automaton (initial states in the reversed automaton).
67 */
68 void insert_initial_lengths();
69
70 /**
71 * Computes shortest words for all states in the automaton.
72 */
73 void compute();
74
75 /**
76 * Computes shortest words for the given @p state.
77 * @param[in] state State to compute shortest words for.
78 */
79 void compute_for_state(State state);
80
81 /**
82 * Creates default shortest words mapping for yet unprocessed @p state.
83 * @param[in] state State to map default shortest words.
84 * @return Created default shortest words map element for the given @p state.
85 */
86 LengthWordsPair map_default_shortest_words(const State state) {
87 return shortest_words_map.emplace(state, std::make_pair(-1, std::set<Word>{})).first->second;
88 }
89
90 /**
91 * Update words for the current state.
92 * @param[out] act Current state shortest words and length.
93 * @param[in] dst Transition target state shortest words and length.
94 * @param[in] symbol Symbol to update with.
95 */
96 static void update_current_words(LengthWordsPair& act, const LengthWordsPair& dst, Symbol symbol);
97}; // Class ShortestWordsMap.

346/**
347 * @brief Create noodles for left and right side of equation.
348 *
349 * Segment automaton is a chain of finite automata (segments) connected via ε-transitions.
350 * A noodle is a copy of the segment automaton with exactly one ε-transition between each two consecutive segments.
351 *
352 * Mata cannot work with equations, queries etc. Hence, we compute the noodles for the equation, but represent
353 * the equation in a way that libMata understands. The left side automata represent the left side of the equation
354 * and the right automaton represents the right side of the equation. To create noodles, we need a segment automaton
355 * representing the intersection. That can be achieved by computing a product of both sides. First, the left side
356 * has to be concatenated over an epsilon transitions into a single automaton to compute the intersection on, though.
357 *
358 * @param[in] lhs_automata Sequence of segment automata for left side of an equation to noodlify.
359 * @param[in] rhs_automaton Segment automaton for right side of an equation to noodlify.
360 * @param[in] include_empty Whether to also include empty noodles.
361 * @param[in] params Additional parameters for the noodlification:
362 * - "reduce": "false", "forward", "backward", "bidirectional"; Execute forward, backward or bidirectional simulation
363 * minimization before noodlification.
364 * @return A list of all (non-empty) noodles.
365 */
366std::vector<Noodle> noodlify_for_equation(const std::vector<std::reference_wrapper<Nfa>>& lhs_automata,
367 const Nfa& rhs_automaton,
368 bool include_empty = false, const ParameterMap& params = {{ "reduce", "false"}});
