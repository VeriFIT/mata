23inline void get_elements(StateSet* element_set, const BoolVector& bool_vec) {
24 element_set->clear();
25 element_set->reserve(bool_vec.count());
26 for (size_t i{ 0 }; i < bool_vec.size(); ++i) {
27 if (bool_vec[i] == 1) {

67/**
68 * @brief Compute intersection of two NFTs.
69 *
70 * Both automata can contain ε-transitions. Epsilons will be handled as alphabet symbols.
71 *
72 * Transducers must share alphabets. //TODO: this is not implemented yet.
73 * Transducers must have equal values of @c num_of_levels.
74 *
75 * @param[in] lhs First NFT to compute intersection for.
76 * @param[in] rhs Second NFT to compute intersection for.
77 * @param[out] prod_map Mapping of pairs of the original states (lhs_state, rhs_state) to new product states (not used internally, allocated only when !=nullptr, expensive).
78 * @param[in] jump_mode Specifies if the symbol on a jump transition (a transition with a length greater than 1)
79 * is interpreted as a sequence repeating the same symbol or as a single instance of the symbol followed by a sequence of @c DONT_CARE
80 * @param[in] lhs_first_aux_state The first auxiliary state in @p lhs. Two auxiliary states does not form a product state.
81 * @param[in] rhs_first_aux_state The first auxiliary state in @p rhs. Two auxiliary states does not form a product state.
82 * @return NFT as a product of NFTs @p lhs and @p rhs.
83 */
84inline void intersection(Nft* res, const Nft& lhs, const Nft& rhs,
85 std::unordered_map<std::pair<State, State>, State> *prod_map = nullptr, const JumpMode jump_mode = JumpMode::RepeatSymbol,
86 const State lhs_first_aux_state = Limits::max_state, const State rhs_first_aux_state = Limits::max_state) {
87 *res = intersection(lhs, rhs, prod_map, jump_mode, lhs_first_aux_state, rhs_first_aux_state);
