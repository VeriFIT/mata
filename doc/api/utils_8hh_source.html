51/// Representation of bool vector by a vector of uint8_t.
52class BoolVector : public std::vector<uint8_t> {
53public:
54 BoolVector(size_t size, bool value) : std::vector<uint8_t>(size, value ? 1 : 0) {}
55 BoolVector(const BoolVector&) = default;
56 BoolVector(BoolVector&&) noexcept = default;
57 BoolVector() = default;
58 BoolVector(std::initializer_list<uint8_t> uint8_ts): std::vector<uint8_t>(uint8_ts) {}
59 explicit BoolVector(const std::vector<uint8_t>& uint8_ts): std::vector<uint8_t>(uint8_ts) {}
60
61 BoolVector& operator=(const BoolVector&) = default;
62 BoolVector& operator=(BoolVector&&) noexcept = default;
63
64 /// Count the number of set elements.
65 size_t count() const {
66 size_t cnt{ 0 };
67 for (const uint8_t value : *this) {
68 if (value == 1) {
69 ++cnt;
70 }
71 }
72 return cnt;
73 }

75 template<typename T>
76 T& get_elements(T& element_set) {
77 element_set.clear();
78 element_set.resize(count());
79 for (size_t i{ 0 }; i < size(); ++i) {
80 element_set.push_back(i);
81 }
82 return element_set;
83 }

85 template<typename T>
86 static T* get_elements(T* element_set, const BoolVector& bool_vec) {
87 element_set->clear();
88 element_set->reserve(bool_vec.count());
89 for (size_t i{ 0 }; i < bool_vec.size(); ++i) {
90 element_set->push_back(i);
91 }
92 return element_set;
93 }

114/**
115 * @brief Replace all occurrences of a substring in a string
116 *
117 * @param input The input string
118 * @param needle The substring to be replaced
119 * @param replace The replacement string
120 * @return A new string with all occurrences of the substring replaced.
121 */
122inline std::string replace_all(const std::string& input, const std::string& needle, const std::string& replace) {
123 std::string result = input;
124 size_t pos = 0;
125 while ((pos = result.find(needle, pos)) != std::string::npos) {
126 result.replace(pos, needle.length(), replace);
127 pos += replace.length();
128 }
129 return result;
130}

132/** Are two sets disjoint? */
133template <class T>
134bool are_disjoint(const std::set<T>& lhs, const std::set<T>& rhs)
135{ // {{{
136 auto itLhs = lhs.begin();
137 auto itRhs = rhs.begin();
138 while (itLhs != lhs.end() && itRhs != rhs.end())
139 {
140 if (*itLhs == *itRhs) { return false; }
141 else if (*itLhs < *itRhs) { ++itLhs; }
142 else {++itRhs; }
143 }
144
145 return true;

155/**
156 * @brief Combine two hash values
157 *
158 * Values taken from
159 * http://www.boost.org/doc/libs/1_64_0/boost/functional/hash/hash.hpp
160 *
161 * TODO: fix to be more suitable for 64b
162 */
163template <class T>
164inline size_t hash_combine(size_t lhs, const T& rhs)
165{ // {{{
166 size_t rhs_hash = std::hash<T>{}(rhs);
167 lhs ^= rhs_hash + 0x9e3779b9 + (lhs<<6) + (lhs>>2);
168 return lhs;
169} // hash_combine }}}

172/**
173 * @brief Hashes a range
174 *
175 * Inspired by
176 * http://www.boost.org/doc/libs/1_64_0/boost/functional/hash/hash.hpp
177 */
178template <typename It>
179size_t hash_range(It first, It last)
180{ // {{{
181 size_t accum = 0;
182
183 for (; first != last; ++first)
184 {
185 accum = hash_combine(accum, *first);
186 }
187
188 return accum;
189} // hash_range(It, It) }}}

199template <template <class, class, class...> class Map, class T1, class T2, class... Args>
200Map<T2, T1> invert_map(const Map<T1, T2>& mp)
201{ // {{{
202 Map<T2, T1> result;
203
204 for (const auto& key_val_pair : mp)
205 {
206 auto it_ins_pair = result.insert({key_val_pair.second, key_val_pair.first});

241// Assuming the doubling extension strategy, it only makes the first reserve large, after that it leaves it to the doubling.
242// Might be worth thinking about it.

244// more than 50% for simple revert,
245// (when testing on a stupid test case)
246template<class Vector>
247void inline reserve_on_insert(Vector & vec,size_t needed_capacity = 0,size_t extension = 32) {
248 //return; //Try this to see the effect of calling this. It should not affect functionality.
249 if (vec.capacity() < extension) //if the size is already large enough, leave it to the default doubling strategy. This if seems to make a barely noticeable difference :).

251 if (vec.capacity() < std::max(vec.size() + 1, needed_capacity))
252 vec.reserve(vec.size() + extension);

256//This function reindexes vector, that is, the content of each index i will be moved to the index renaming[i].
257// It might be useful in revert and trim, but so far it is useless. It was hard to get right, so I am reluctant to remove ....
258// It assumes that renaming[i] <= i.
259// It assumes that vec is not longer than renaming.
260// The function is very fragile.
261template<class Vector,typename Index>
262void defragment(Vector & vec, const std::vector<Index> & renaming) {
263 //assert(vec.size() <= renaming.size());
264 size_t i = 0;
265 for (size_t rsize=renaming.size(),vsize=vec.size(); i<vsize && i<rsize ; i++) {
266 if (renaming[i] != i)
267 {
268 if(! (renaming[i]<vsize) )
269 break;
270 assert(renaming[i] < i);
271 vec[i] = std::move(vec[renaming[i]]);
272 }
273 }
274 vec.reserve(i);
275}
