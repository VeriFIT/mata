69/**
70 * @brief Compute intersection of two NFAs.
71 *
72 * Both automata can contain ε-transitions. The product preserves the ε-transitions, i.e.,
73 * for each each product state `(s, t)` with`s -ε-> p`, `(s, t) -ε-> (p, t)` is created, and vice versa.
74 *
75 * Automata must share alphabets.
76 *
77 * @param[out] res The resulting intersection NFA.
78 * @param[in] lhs Input NFA.
79 * @param[in] rhs Input NFA.
80 * @param[in] first_epsilon smallest epsilon.
81 * @param[out] prod_map Mapping of pairs of the original states (lhs_state, rhs_state) to new product states (not used internally, allocated only when !=nullptr, expensive).
82 * @return NFA as a product of NFAs @p lhs and @p rhs with ε-transitions preserved.
83 */
84inline void intersection(Nfa* res, const Nfa& lhs, const Nfa& rhs, Symbol first_epsilon = EPSILON,
85 std::unordered_map<std::pair<State, State>, State> *prod_map = nullptr) {
86 *res = intersection(lhs, rhs, first_epsilon, prod_map);
87}
