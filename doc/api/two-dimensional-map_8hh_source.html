14/**
15 * @brief A two-dimensional map that can be used to store pairs of values and their associated value.
16 * This class can handle both small and large maps efficiently.
17 *
18 * The largest matrix of pairs of we are brave enough to allocate is 50'000'000 for 8 Byte values.
19 * So ten million cells is close to 100 MB.
20 * If the number is larger, then we do not allocate a matrix, but use a vector of unordered maps (vec_map_storage).
21 * The unordered_map seems to be about twice slower.
22 *
23 * @tparam T Type of the values stored in the map. Must be an unsigned type.
24 * @tparam track_inverted Whether to track inverted indices for the first and second dimensions.
25 * To use this feature correctly, the mapping has to be reversible.
26 * @tparam MAX_MATRIX_SIZE Maximum size of the matrix before switching to vector of unordered maps.
27 */
28template <typename T, bool track_inverted = true, size_t MAX_MATRIX_SIZE = 50'000'000>
29class TwoDimensionalMap {
30 static_assert(std::is_unsigned<T>::value, "ProductStorage requires an unsigned type");
31
32public:
33 using Map = std::unordered_map<std::pair<T, T>, T>;
34 using MatrixStorage = std::vector<std::vector<T>>;
35 using VecMapStorage = std::vector<std::unordered_map<T, T>>;
36 using InvertedStorage = std::vector<T>;
37
38 /**
39 * @brief Constructor for TwoDimensionalMap.
40 * @param first_dim_size Size of the first dimension.
41 * @param second_dim_size Size of the second dimension.
42 */
43 TwoDimensionalMap(const size_t first_dim_size, const size_t second_dim_size)
44 : is_large(first_dim_size * second_dim_size > MAX_MATRIX_SIZE),
45 first_dim_size(first_dim_size), second_dim_size(second_dim_size)
46 {
47 assert(first_dim_size < std::numeric_limits<T>::max());
48 assert(second_dim_size < std::numeric_limits<T>::max());
49 if (!is_large) {
50 matrix_storage = MatrixStorage(first_dim_size, InvertedStorage(second_dim_size, std::numeric_limits<T>::max()));
51 } else {
52 vec_map_storage = VecMapStorage(first_dim_size);
53 }
54 if constexpr (track_inverted) {
55 first_dim_inverted.resize(first_dim_size + second_dim_size);
56 second_dim_inverted.resize(first_dim_size + second_dim_size);
57 }
58 }
59
60 /**
61 * @brief Get the value associated with a pair of keys.
62 * @param first First key.
63 * @param second Second key.
64 * @return The value associated with the pair, or std::numeric_limits<T>::max() if not found.
65 */
66 T get(const T first, const T second) const {
67 if (!is_large) {
68 return matrix_storage[first][second];
69 }
70
71 auto it = vec_map_storage[first].find(second);
72 if (it == vec_map_storage[first].end()) {
73 return std::numeric_limits<T>::max();
74 }
75 return it->second;
76 }
77
78 /**
79 * @brief Insert a value associated with a pair of keys.
80 * @param first First key.
81 * @param second Second key.
82 * @param value Value to associate with the pair.
83 */
84 void insert(const T first, const T second, const T value) {
85 if (!is_large) {
86 matrix_storage[first][second] = value;
87 } else {
88 vec_map_storage[first][second] = value;
89 }
90 if constexpr (track_inverted) {
91 first_dim_inverted.resize(value + 1);
92 second_dim_inverted.resize(value + 1);
93 first_dim_inverted[value] = first;
94 second_dim_inverted[value] = second;
95 }
96 }
97
98 /**
99 * @brief Get the first inverted index for a @p value.
100 * This is only available if track_inverted is true.
101 */
102 T get_first_inverted(const T value) const {
103 static_assert(track_inverted, "get_first_inverted only available if track_inverted is true");
104 return first_dim_inverted[value];
105 }
106
107 /**
108 * @brief Get the second inverted index for a @p value.
109 * This is only available if track_inverted is true.
110 */
111 T get_second_inverted(const T value) const {
112 static_assert(track_inverted, "get_second_inverted only available if track_inverted is true");
113 return second_dim_inverted[value];
114 }
115
116private:
117 const bool is_large;
118 const size_t first_dim_size;
119 const size_t second_dim_size;
120 MatrixStorage matrix_storage{};
121 std::vector<std::unordered_map<T, T>> vec_map_storage{};
122 InvertedStorage first_dim_inverted{};
123 InvertedStorage second_dim_inverted{};
124
125};
