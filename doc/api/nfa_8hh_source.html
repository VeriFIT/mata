611 /**
612 * @brief Make NFA complete in place.
613 *
614 * For each state 0,...,this->num_of_states()-1, add transitions with "missing" symbols from @p alphabet
615 * (symbols that do not occur on transitions from given state) to @p sink_state. If @p sink_state does not belong
616 * to the NFA, it is added to it, but only in the case that some transition to @p sink_state was added.
617 * In the case that NFA does not contain any states, this function does nothing.
618 *
619 * @param[in] alphabet Alphabet to use for computing "missing" symbols. If @c nullptr, use @c this->alphabet when
620 * defined, otherwise use @c this->delta.get_used_symbols().

656// Using parameter pack and variadic arguments.
657// Adapted from: https://www.fluentcpp.com/2019/01/25/variadic-number-function-parameters-type/.
658/// Pack of bools for reasoning about a sequence of parameters.
659template<bool...> struct bool_pack{};
